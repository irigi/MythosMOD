		FROM = { trait = vampire }
	}
	
	immediate = {
		add_trait = vampire
		dynasty = FROM	
		IF = {
			limit = { FROM = { is_female = yes } }
			set_mother = FROM
			set_father = none
		}
		IF = {
			limit = { FROM = { is_female = no } }
			set_father = FROM
			set_mother = none
		}
		if = {
			limit = { NOT = { has_opinion_modifier = { modifier = opinion_former_ghoul who = FROM } } }
			opinion = { # is not ghoul, but Embrace creates residual blood bond that vanishes eventually
				modifier = opinion_former_ghoul
				who = FROM
			}
		}
		remove_opinion = {
			modifier = opinion_my_ghoul
			who = FROM
		}	
		if {
			limit = { trait = ghoul }
			change_variable = { which = "ghouls_count" value = -1 }
		}		
		remove_trait = ghoul				
		clr_character_flag = flag_ghoul_has_master		
		remove_character_modifier = ghoul_without_master		
	}

	option = {
		name = "EVTOPTAMYTH.84"	
		add_trait = vampire
	}
}
